It returns the current breakpoint value.

<Tabs {...defaultTabsProps}>
<TabItem value="typescript">

```tsx
declare const useCurrentBreakpoint: () => 'desktop' | 'mobile' | 'tablet'
```

</TabItem>
<TabItem value="rescript">

```res
type breakpoint = [#mobile | #tablet | #desktop]

let useCurrentBreakpoint: unit => breakpoint
```

</TabItem>
</Tabs>

## Example

```jsx live
function() {
  const DebugText = () => {
    const currentBreakpoint = useCurrentBreakpoint()

    return (
      <Text>{currentBreakpoint}</Text>
    )
  }

  return (
    <StacksProvider>
      <DebugText />
    </StacksProvider>
  )
}
```
