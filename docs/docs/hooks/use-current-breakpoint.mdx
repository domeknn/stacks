It returns the current breakpoint value.

<Tabs {...defaultTabsProps}>

```tsx
declare const useCurrentBreakpoint: () => string
```

</TabItem>
<TabItem value="rescript">

</TabItem>
</Tabs>

## Example

```jsx live
function() {
  const DebugText = () => {
    const currentBreakpoint = useCurrentBreakpoint()

    return (
      <Text>{currentBreakpoint}</Text>
    )
  }

  return (
    <StacksProvider breakpoints={[
      ['mobile', 480],
      ['tablet', 768],
      ['desktop', 992],
      ['large-desktop', 1200],
      ['extra-large-desktop', 1440]
    ]}>
      <DebugText />
    </StacksProvider>
  )
}
```
