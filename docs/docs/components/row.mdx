import { Props, BoxProps } from '../../core/components';

Use the `Row` component along with `Rows`.

:::note

`Column` component accepts [`View`](https://reactnative.dev/docs/view) props.

:::

## Examples

```jsx live
function() {
  return (
    <StacksProvider>
      <PlaceholderView height={500}>
        <Rows space={1} alignY="evenly">
          <Row height="content">
            <Placeholder height={100} />
          </Row>
          <Row height="1/2">
            <FluidPlaceholder />
          </Row>
          <Row height="content">
            <Placeholder height={100} />
          </Row>
        </Rows>
      </PlaceholderView>
    </StacksProvider>
  )
}
```

```jsx live
function() {
  return (
    <StacksProvider>
      <PlaceholderView height={500}>
        <Rows space={4}>
          <Row>
            <FluidPlaceholder />
          </Row>
          <Row height="content">
            <Columns space={4}>
              <Column>
                <Placeholder height={200} />
              </Column>
              <Column>
                <Placeholder height={200} />
              </Column>
            </Columns>
          </Row>
          <Row>
            <FluidPlaceholder />
          </Row>
        </Rows>
      </PlaceholderView>
    </StacksProvider>
  )
}

```
