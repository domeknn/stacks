`Grid` renders a design system grid. Use it only once, at the top of your react tree.

### Basic usage

```tsx
import { Grid } from '@mobily/stacks'

<Grid columns={8} margin={2} gutter={2} opacity={0.2} />
```

### Props

<Props
  data={[
    {
      name: 'columns',
      type: 'responsiveProp<int>',
      required: false,
      defaultValue: '8',
    },
    {
      name: 'margin',
      type: 'responsiveProp<float>',
      required: false,
      defaultValue: '2',
    },
    {
      name: 'gutter',
      type: 'responsiveProp<float>',
      required: false,
      defaultValue: '2',
    },
    {
      name: 'opacity',
      type: 'float',
      required: false,
      defaultValue: '0.2',
    },
  ]}
/>
