`FillView` covers a common pattern for creating overlays with position absolute and zero positioning.


### Basic usage

```jsx live showCode
// import { reset } from '@mobily/stacks'

const styles = StyleSheet.create({
  root: {
    backgroundColor: '#c2abfa',
  }
})

const App = () => {
  return (
    <Device>
      <FillView alignX="center" alignY="center" style={styles.root}>
        <Image name="book" width={128} />
      </FillView>
    </Device>
  )
}

render(<App />)
```

### Reset


```jsx live
// import { reset } from '@mobily/stacks'

const App = () => {
  return (
    <Device>
      <FillView top={40} left={16} bottom={reset} right={reset}>
        <Placeholder width={80} />
      </FillView>

      <FillView top={40} right={16} bottom={reset} left={reset}>
        <Placeholder width={80} />
      </FillView>

      <FillView bottom={30} left={16} right={16} top={reset}>
        <Columns space={4}>
          <Placeholder />
          <Placeholder />
          <Placeholder />
        </Columns>
      </FillView>
    </Device>
  )
}

render(<App />)
```
